    import sys
import re

def GetTheSentences():
  for result in re.findall('::: {#ref-(.*?):::', sys.stdin.read(), re.S):
     entry = result.split("\n")
     reference = f"[#{entry[0].split()[0]}]:"
     citation = "\n".join(entry[1:])
     print(f"{reference} {citation}")

if __name__ == "__main__":
    GetTheSentences()
